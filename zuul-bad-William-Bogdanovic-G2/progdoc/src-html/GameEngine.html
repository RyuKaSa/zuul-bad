216    /**
217     * Given a command, process (that is: execute) the command.
218     * If this command ends the game, true is returned, otherwise false is
219     * returned.
220     * 
221     * @param pCommandLine to read and the actions of the buttons
222     */
223    public void interpretCommand(final String pCommandLine) 
224    {
225        this.aGui.println( "> " + pCommandLine );
226        Command vCommand = this.aParser.getCommand( pCommandLine);
227
228        if(vCommand.isUnknown()) {
229            this.aGui.println("I don't know what you mean...");
230            return;
231        }
232
233        String vCommandWord = vCommand.getCommandWord();
234        if(vCommandWord.equals("help"))
235            this.printHelp();
236        else if(vCommandWord.equals("go"))
237            this.goRoom(vCommand);
238        else if(vCommandWord.equals("eat"))
239            this.aPlayer.eat(vCommand);
240        else if(vCommandWord.equals("look"))
241            this.aPlayer.look(vCommand);
242        else if(vCommandWord.equals("back"))
243            this.back(vCommand);
244        else if(vCommandWord.equals("test"))
245            this.test(vCommand); 
246        else if(vCommandWord.equals("take"))
247            this.aPlayer.take(vCommand); 
248        else if(vCommandWord.equals("drop"))
249            this.aPlayer.drop(vCommand);
250        else if(vCommandWord.equals("items"))
251            this.aPlayer.items(/*vCommand*/);
252        else if(vCommandWord.equals("charge"))
253            this.aPlayer.charge(/*vCommand*/);
254        else if(vCommandWord.equals("fire"))
255            this.aPlayer.fire(/*vCommand*/);
256        else if(vCommandWord.equals("map"))
257            this.aPlayer.map();
258        else if(vCommandWord.equals("quit")){
259            if(vCommand.hasSecondWord())
260                this.aGui.println("Quit what?");
261            else
262                this.endGame();
263        }
264    }//interpretCommand(.)

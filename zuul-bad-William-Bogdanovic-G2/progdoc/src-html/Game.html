002/**
003 *  This class is the main class of the "World of Zuul" application. 
004 *  "World of Zuul" is a very simple, text based adventure game.  Users 
005 *  can walk around some scenery. That's all. It should really be extended 
006 *  to make it more interesting!
007 * 
008 *  To play this game, create an instance of this class.
009 * 
010 *  This main class creates the necessary implementation objects and starts the game off.
011 * 
012 * @author  Michael Kolling and David J. Barnes and Bogdanovic William
013 * @version 2.0 (Jan 2003) DB edited (2019)
014 */
015
016public class Game
017{
018    private UserInterface aGui;
019    private GameEngine aEngine;
020
021    /**
022     * Create the game and initialise its internal map. Create the inerface and link to it.
023     * 
024     */
025    public Game() 
026    {
027        this.aEngine = new GameEngine();
028        this.aGui = new UserInterface(this.aEngine);
029        this.aEngine.setGUI(this.aGui);
030    }//Game()
031    
032    
033}//Game

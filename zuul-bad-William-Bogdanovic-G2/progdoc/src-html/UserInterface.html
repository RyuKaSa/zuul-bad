049    /**
050     * Print out some text into the text area.
051     * 
052     * @param pText that we want printed in the text area
053     */
054    public void print( final String pText )
055    {
056        this.aLog.append( pText );
057        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
058    }//print(.)
059
060    /**
061     * Print out some text into the text area, followed by a line break.
062     * 
063     * @param pText that we want printed in the text area on seperate lines (\n)
064     */
065    public void println( final String pText )
066    {
067        this.print( pText + "\n" );
068    }//println(.)
069
070    /**
071     * Show an image file in the interface.
072     * 
073     * @param pImageName of the image name linked to its room
074     */
075    public void showImage( final String pImageName )
076    {
077        String vImagePath = "" + pImageName; // to change the directory
078        URL vImageURL = this.getClass().getClassLoader().getResource( vImagePath );
079        if ( vImageURL == null )//code to run if there is no image found
080            this.aGui.println( "Image not found : " + vImagePath );
081        else {//display the image
082            ImageIcon vIcon = new ImageIcon( vImageURL );
083            this.aImage.setIcon( vIcon );
084            this.aMyFrame.pack();
085        }
086    }//showImage(.)
087
088    /**
089     * Enable or disable input in the input field.
090     * 
091     * @param pOnOff to disable or not the entry field when needed
092     */
093    public void enable( final boolean pOnOff )
094    {
095        this.aEntryField.setEditable(pOnOff);//enable/disable
096        if (!pOnOff) { // disable
097            this.aEntryField.getCaret().setBlinkRate(0);//cursor won't blink
098            this.aEntryField.removeActionListener(this);//won't react to entry
099        }
100    }//enable(.)

102    /**
103     * Set up graphical user interface.
104     * 
105     */
106    private void createGUI()
107    {
108        this.aMyFrame = new JFrame("JailBreak Dungeon"); //title
109        this.aEntryField = new JTextField(34);
110
111        this.aLog = new JTextArea();
112        this.aLog.setEditable(false);
113        JScrollPane vListScroller = new JScrollPane( this.aLog );
114        vListScroller.setPreferredSize( new Dimension(310, 100) );
115        vListScroller.setMinimumSize( new Dimension(310, 100) );
116
117        JPanel vPanel = new JPanel();
118        this.aImage = new JLabel();
119
120        vPanel.setLayout( new BorderLayout() ); // ==> only five places
121        vPanel.add( this.aImage, BorderLayout.NORTH );
122        vPanel.add( vListScroller, BorderLayout.CENTER );
123        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
124        vPanel.add( this.gridButtonUtility(), BorderLayout.WEST );
125        vPanel.add( this.gridButtonDirection(), BorderLayout.EAST );
126        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.CENTER );
127
128        // add some event listeners to some components
129        this.aEntryField.addActionListener( this );
130        
131        // to end program when window is closed
132        this.closeWindow();
133        /*this.aMyFrame.addWindowListener( new WindowAdapter() {
134            public void windowClosing(WindowEvent e) { System.exit(0); }
135        } );*/
136        
137        this.aMyFrame.pack();
138        this.aMyFrame.setVisible( true );
139        this.aEntryField.requestFocus();
140    }//createGUI()

363    /**

367     */

489     * A command has been entered. Read the command and do whatever is 
490     * necessary to process it.
491     * 
492     */
493    private void processCommand()
494    {

497
498        this.aEngine.interpretCommand( vInput );
